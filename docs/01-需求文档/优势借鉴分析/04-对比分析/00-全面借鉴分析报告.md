### 3.1 Auto Mode 状态机 (来自 GSD-2)

**借鉴价值**: ⭐⭐⭐⭐⭐ (最高)

**现状问题**:
- Spec-First 每步需要手动触发命令
- 无自动循环执行能力
- 用户无法"离开"

**GSD-2 实现参考**:
```typescript
// gsd-2/src/resources/extensions/gsd/auto.ts (3,463 行)

while (active && !paused) {
  const state = deriveState(basePath);  // 从磁盘派生状态

  switch (state.phase) {
    case 'research':
      await dispatchResearch(state);
      break;
    case 'plan_slice':
      await dispatchPlanSlice(state);
      break;
    case 'execute_task':
      await dispatchExecuteTask(state);
      break;
    case 'complete_slice':
      await dispatchCompleteSlice(state);
      break;
  }

  // 自动循环，无需用户干预
}
```

**Spec-First 实现建议**:
```typescript
// spec-first/src/core/auto-loop/index.ts (新增)

export class AutoLoopEngine {
  private active = false;
  private paused = false;

  async start(featureId: string): Promise<void> {
    this.active = true;

    while (this.active && !this.paused) {
      const state = loadFeatureState(featureId);

      switch (state.stage) {
        case '01_specify':
          await this.dispatchSpecPhase(featureId);
          break;
        case '02_design':
          await this.dispatchDesignPhase(featureId);
          break;
        case '03_plan':
          await this.dispatchPlanPhase(featureId);
          break;
        case '04_implement':
          await this.dispatchImplementPhase(featureId);
          break;
        case '05_verify':
          await this.dispatchVerifyPhase(featureId);
          break;
      }

      // Gate 校验
      const gateResult = await checkGate(featureId, state.stage);
      if (!gateResult.passed) {
        this.pause();
        reportGateFailure(gateResult);
      }
    }
  }
}
```

**CLI 命令**:
```bash
spec-first auto --feature FSREQ-001    # 启动自动模式
spec-first auto --stop                 # 停止
spec-first auto --pause                # 暂停
spec-first auto --resume               # 恢复
```

**Spec-First 实现建议**:
```typescript
// spec-first/src/core/metrics-engine/cost-tracker.ts (新增)

export interface FeatureMetrics {
  featureId: string;
  totalCost: number;
  totalTokens: number;
  stageBreakdown: Map<Stage, StageMetrics>;
  budgetCeiling?: number;
}

export interface StageMetrics {
  inputTokens: number;
  outputTokens: number;
  cost: number;
  model: string;
  unitCount: number;
}

export class CostTracker {
  trackUnitExecution(unit: UnitExecution): void {
    const metrics = this.calculateMetrics(unit);
    this.persistMetrics(metrics);

    if (this.exceedsBudget(metrics)) {
      this.emitBudgetWarning();
    }
  }

  generateDashboard(): DashboardReport {
    return {
      featureId: this.featureId,
      totals: this.getTotals(),
      stageBreakdown: this.getStageBreakdown(),
      projections: this.calculateProjections(),
      budgetStatus: this.getBudgetStatus(),
    };
  }
}
```

### 3.5 三层超时监督 (来自 GSD-2)

**借鉴价值**: ⭐⭐⭐⭐

**现状问题**:
- 无超时监督
- 任务可能无限运行
- 无法检测停滞

**GSD-2 实现**:
```typescript
// soft_timeout_minutes: 20 - 警告 LLM 收尾
// idle_timeout_minutes: 10 - 检测停滞
// hard_timeout_minutes: 30 - 强制暂停

unitTimeoutHandle = setTimeout(() => {
  injectTimeoutWarning("Please wrap up within 5 minutes");
}, config.softTimeoutMinutes * 60 * 1000);

idleWatchdogHandle = setInterval(() => {
  if (noProgress()) {
    handleIdle("No progress detected for 10 minutes");
  }
}, config.idleTimeoutMinutes * 60 * 1000);
```

**Spec-First 实现建议**:
```typescript
// spec-first/src/core/ai-orchestrator/timeout-supervisor.ts (新增)

export class TimeoutSupervisor {
  private softTimeout: number;   // 20 min
  private idleTimeout: number;   // 10 min
  private hardTimeout: number;   // 30 min

  startMonitoring(execution: TaskExecution): void {
    // Soft timeout: 提示收尾
    this.scheduleSoftWarning(() => {
      this.injectWarning("Approaching timeout. Please wrap up current work.");
    });

    // Idle watchdog: 检测停滞
    this.startIdleWatchdog(() => {
      if (this.noProgress(execution)) {
        this.pauseWithDiagnostic("Idle detected. Generating diagnostic...");
      }
    });

    // Hard timeout: 强制暂停
    this.scheduleHardTimeout(() => {
      this.forcePause("Hard timeout reached. Saving state...");
    });
  }

  private noProgress(execution: TaskExecution): boolean {
    // 检查是否有工具调用、文件写入等
    return execution.lastActivity > this.idleTimeout;
  }
}
```

**配置**:
```yaml
# .spec-first/preferences.yaml
timeout:
  soft_minutes: 20    # 警告收尾
  idle_minutes: 10     # 检测停滞
  hard_minutes: 30     # 强制暂停
```

```typescript
// spec-first/src/core/git-automation/index.ts (新增)

export class GitAutomation {
  async commitTask(task: Task): Promise<void> {
    await this.stageFiles(task.changedFiles);
    await this.commit(`feat(${task.id}): ${task.title}`);
  }

  async createFeatureBranch(featureId: string): Promise<void> {
    await this.run(['checkout', '-b', `feature/${featureId}`]);
  }

  async mergeFeatureBranch(featureId: string): Promise<void> {
    await this.run(['checkout', this.mainBranch]);
    await this.run(['merge', '--squash', `feature/${featureId}`]);
    await this.run(['commit', '-m', this.generateMergeMessage(featureId)]);
    await this.run(['branch', '-D', `feature/${featureId}`]);
  }
}
```

**Spec-First 增强**:
```typescript
// spec-first/src/core/batch-executor/wave-scheduler.ts (增强)

export class WaveScheduler {
  scheduleByDependencies(tasks: Task[]): Wave[] {
    const graph = buildDependencyGraph(tasks);
    return topologicalSortIntoWaves(graph);
  }

  async executeWave(wave: Wave): Promise<WaveResult> {
    return Promise.all(wave.tasks.map(t => executeTask(t)));
  }

  async executeAllWaves(waves: Wave[]): Promise<WaveResult[]> {
    const results = [];
    for (const wave of waves) {
      const waveResult = await this.executeWave(wave);
      results.push(waveResult);

      // 检查依赖是否满足
      if (!waveResult.success) {
        await this.handleWaveFailure(wave, waveResult);
      }
    }
    return results;
  }
}
```
