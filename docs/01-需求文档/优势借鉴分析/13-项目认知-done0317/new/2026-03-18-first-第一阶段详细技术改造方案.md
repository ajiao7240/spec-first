**目标逻辑**：
```typescript
export function handleFirst(args: string[]): number {
  const projectRoot = process.cwd();
  const firstArgs = validateFirstArgs(args);
  
  // 健康检查模式
  if (firstArgs.checkHealth) {
    return handleCheckHealth(projectRoot);
  }
  
  // 正常执行模式
  const index = readFirstRuntimeIndex(projectRoot);
  const hasHealthyRuntime = isRuntimeHealthy(index);
  
  if (hasHealthyRuntime && !firstArgs.force) {
    // 有健康的 runtime → 刷新
    const result = refreshFirstArtifacts(projectRoot);
    console.log(`✓ 已刷新 first runtime 与 docs`);
  } else {
    // 无 runtime 或强制重新生成 → bootstrap
    const result = bootstrapFirstRuntime(projectRoot, {
      platformType: firstArgs.type,
    });
    console.log(`✓ 已生成 first runtime 与 docs`);
  }
  
  return ExitCode.SUCCESS;
}
```

**改动点**：
1. 移除 `skip` 分支
2. 简化执行逻辑为两路径：bootstrap / refresh
3. 统一输出格式
