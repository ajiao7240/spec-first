```typescript
// tech stack parser
async function collectTechStack(projectRoot: string, isTTY: boolean): Promise<string> {
  if (isTTY) {
    return await question('请描述技术栈 (如: react+typescript+node): ');
  }
  return ''; // 非 TTY 时返回空字符串，由调用方决定是否继续
}

// platform template inference
function inferPlatformTemplates(techStack: string): string[] {
  const templates: string[] = [];
  const lower = techStack.toLowerCase();
  if (lower.includes('react')) templates.push('platform-web-react');
  if (lower.includes('vue')) templates.push('platform-web-vue');
  if (lower.includes('node') || lower.includes('express')) templates.push('platform-node-express');
  if (lower.includes('spring')) templates.push('platform-java-spring');
  // ... 更多推断规则
  return templates.length > 0 ? templates : ['platform-generic'];
}
```
