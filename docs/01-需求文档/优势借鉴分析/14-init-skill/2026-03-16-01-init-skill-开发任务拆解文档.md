### 扩展点 3：配置化推断规则

**阶段 1 可选实现**：

**配置文件格式**（`.spec-first/platform-mappings.json`）：

```json
{
  "version": "1.0",
  "mappings": [
    {
      "keywords": ["react", "reactjs", "jsx"],
      "template": "platform-web-react",
      "priority": 10
    },
    {
      "keywords": ["vue", "vuejs", "vuex"],
      "template": "platform-web-vue",
      "priority": 10
    },
    {
      "keywords": ["rust", "rust-lang", "cargo"],
      "template": "platform-rust-axum",
      "priority": 8
    }
  ],
  "defaultTemplate": "platform-generic"
}
```

**代码实现**：

```typescript
import { readFileSync } from 'node:fs';
import { join } from 'node:path';

interface PlatformMapping {
  keywords: string[];
  template: string;
  priority: number;
}

interface PlatformMappingsConfig {
  version: string;
  mappings: PlatformMapping[];
  defaultTemplate: string;
}

function loadPlatformMappings(projectRoot: string): PlatformMappingsConfig {
  const configPath = join(projectRoot, '.spec-first/platform-mappings.json');

  try {
    const content = readFileSync(configPath, 'utf-8');
    return JSON.parse(content) as PlatformMappingsConfig;
  } catch {
    // 配置文件不存在，返回硬编码默认值
    return {
      version: '1.0',
      mappings: [/* 硬编码的默认映射 */],
      defaultTemplate: 'platform-generic',
    };
  }
}

function inferPlatformTemplates(techStack: string): string[] {
  const config = loadPlatformMappings(projectRoot);
  const lower = techStack.toLowerCase();

  const matchedTemplates = config.mappings
    .filter(mapping => mapping.keywords.some(kw => lower.includes(kw)))
    .sort((a, b) => b.priority - a.priority)
    .map(mapping => mapping.template);

  return matchedTemplates.length > 0 ? matchedTemplates : [config.defaultTemplate];
}
```

**优势**：
- 用户可以自定义推断规则，无需修改代码
- 支持优先级（priority 字段）
- 向后兼容（配置文件缺失时使用硬编码默认值）

**变更流程**（配置化后）：
```
1. 用户编辑 .spec-first/platform-mappings.json
2. 无需修改代码
3. 重启 CLI 即可生效
```

function detectInitTrack(state: InitProjectState, args: string[]): InitTrack {
  // 显式 --track 优先
  const trackArg = args.find(arg => arg.startsWith('--track='));
  if (trackArg) {
    const track = trackArg.split('=')[1] as InitTrack;
    if (['project-onboarding', 'brownfield-baseline', 'feature-init'].includes(track)) {
      return track;
    }
  }

  // 用户传入 --feat 参数，尝试 feature-init（但需验证 first 健康）
  if (args.includes('--feat')) {
    if (!state.firstRuntimeHealthy) {
      throw new InitError('ERR_FIRST_UNHEALTHY',
        'Feature 初始化需要 first runtime，请先运行 project-onboarding 补齐',
        { recovery: 'spec-first init --track project-onboarding' });
    }
    return 'feature-init';
  }

  // 自动路由规则
  if (!state.gitReady) {
    throw new InitError('ERR_NO_GIT', '请先执行 git init 初始化 git 仓库');
  }

  if (!state.specFirstDirExists || !state.metaConfigExists) {
    return 'project-onboarding';
  }

  if (!state.firstRuntimeHealthy) {
    return 'project-onboarding-with-first';
  }

  // B7/B9: first 不健康时，不允许 feature-init
  if (!state.firstRuntimeHealthy) {
    throw new InitError('ERR_FIRST_UNHEALTHY',
      'first runtime 不健康，请修复或重新生成',
      { recovery: 'spec-first doctor && spec-first first' });
  }

  // B5: brownfield 且无 baseline
  if (state.projectMaturity === 'brownfield' && !state.hasLegacyBaseline) {
    return 'brownfield-baseline';
  }

  // 默认路由到 feature-init
  return 'feature-init';
}

// tech stack parser
async function collectTechStack(projectRoot: string, isTTY: boolean): Promise<string> {
  if (isTTY) {
    return await question('请描述技术栈 (如: react+typescript+node): ');
  }
  return ''; // 非 TTY 时返回空字符串，由调用方决定是否继续
}

// platform template inference with conflict detection
interface InferenceResult {
  templates: string[];
  conflicts: string[];  // 冲突的平台描述（如 "react vs vue"）
  hasAmbiguity: boolean;
}

function inferPlatformTemplates(techStack: string): InferenceResult {
  const templates: string[] = [];
  const conflicts: string[] = [];
  const lower = techStack.toLowerCase();

  // 前端框架冲突检测（react vs vue vs angular）
  const frontendMatches: string[] = [];
  if (lower.includes('react')) {
    templates.push('platform-web-react');
    frontendMatches.push('react');
  }
  if (lower.includes('vue')) {
    templates.push('platform-web-vue');
    frontendMatches.push('vue');
  }
  if (lower.includes('angular')) {
    templates.push('platform-web-angular');
    frontendMatches.push('angular');
  }

  // 检测到多个前端框架时标记冲突
  if (frontendMatches.length > 1) {
    conflicts.push(`前端框架冲突：${frontendMatches.join(' vs ')}`);
  }

  // 后端框架推断
  if (lower.includes('node') || lower.includes('express')) {
    templates.push('platform-node-express');
  }
  if (lower.includes('spring')) {
    templates.push('platform-java-spring');
  }

  return {
    templates: templates.length > 0 ? templates : ['platform-generic'],
    conflicts,
    hasAmbiguity: conflicts.length > 0
  };
}
