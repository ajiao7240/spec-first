## 一、核心洞察：数据完整性 vs 业务完整性

### 关键区分

**数据完整性**（Infrastructure Layer）：
- ID 格式合法性
- 引用存在性（引用的 ID 必须存在）
- 链路类型匹配（FR 的 upstream 只能是 REQ）
- **类比**：数据库的外键约束

**业务完整性**（Application Layer）：
- 孤儿检测（非根节点必须有 upstream）
- 断链检测（FR 缺少必需的 DS/TASK/TC）
- 覆盖率要求（阶段门禁）
- **类比**：业务规则校验

### 为什么这样划分？

**场景对比**：

```typescript
// 数据完整性问题（基础设施层）
FR-AUTH-001 → upstream: ["REQ-PRD-001"]  // ❌ REQ-PRD 格式非法
FR-AUTH-002 → upstream: ["DS-AUTH-001"]  // ❌ FR 不能有 DS 作为 upstream

// 业务完整性问题（应用层）
FR-AUTH-003 → upstream: []               // ❌ 孤儿（业务规则）
FR-AUTH-004 → 没有 DS downstream         // ❌ 断链（缺少必需下游）
FR-AUTH-005 → C2 覆盖率 < 80%            // ❌ 覆盖率不足（阶段要求）
```

**关键**：数据完整性问题会导致系统崩溃，必须在入口处拦截。

## 四、Infrastructure Layer（修正版）

### parseMatrix 职责

```typescript
// src/core/trace-engine/matrix.ts
export function parseMatrix(content: string): ParsedMatrix {
  // 1. 格式校验
  // 2. 引用存在性校验
  // 3. 链路类型匹配校验
  
  const errors: string[] = [];
  const idIndex = new Map<string, MatrixRow>();
  
  // 第一遍：构建索引
  for (const row of rows) {
    if (!validateIdFormat(row.id)) {
      errors.push(`ID 格式非法: ${row.id}`);
    }
    idIndex.set(row.id, row);
  }
  
  // 第二遍：校验引用
  for (const row of rows) {
    for (const upId of row.upstream) {
      if (!idIndex.has(upId)) {
        errors.push(`引用不存在: ${row.id} → ${upId}`);
      }
      if (!validateChainType(row.id, upId, 'upstream')) {
        errors.push(`链路类型非法: ${row.id} upstream 不能是 ${upId}`);
      }
    }
  }
  
  if (errors.length > 0) throw new Error(errors.join('\n'));
  return { rows, idIndex };
}
```

## 五、Application Layer（修正版）

### checkMatrix 职责（统一使用 Gate 规则）

```typescript
// src/core/trace-engine/matrix.ts
import { STAGE_GATE_RULES } from '../gate-engine/gate-rules.js';

export function checkMatrix(parsed: ParsedMatrix, stage?: string): CheckResult {
  const orphans: string[] = [];
  const brokenChains: BrokenChain[] = [];

  // 1. 孤儿检测（从 STAGE_GATE_RULES 读取豁免）
  const allowedOrphans = stage ? STAGE_GATE_RULES[stage]?.allowedOrphans || [] : [];

  for (const row of parsed.rows) {
    const idType = getIdType(row.id);
    if (ROOT_ID_TYPES.has(idType)) continue;

    const exempted = allowedOrphans.includes(idType);
    if (!exempted && row.upstream.length === 0) {
      orphans.push(row.id);
    }
  }

  // 2. 断链检测（输出具体缺失）
  for (const row of parsed.rows) {
    const idType = getIdType(row.id);
    const requiredDownstream = getRequiredDownstream(idType, stage);

    for (const reqType of requiredDownstream) {
      const hasDownstream = parsed.rows.some(
        r => getIdType(r.id) === reqType && r.upstream.includes(row.id)
      );
      if (!hasDownstream) {
        brokenChains.push({
          id: row.id,
          missing: reqType,
          reason: `${idType} 缺少必需的 ${reqType} 下游`,
        });
      }
    }
  }

  return { orphans, brokenChains };
}

interface BrokenChain {
  id: string;
  missing: IdType;
  reason: string;
}
```

**关键修正**：
- ✅ 从 STAGE_GATE_RULES.allowedOrphans 读取豁免规则
- ✅ 单一真源，避免重复定义
- ✅ 保留断链检测，输出具体缺失

// src/core/trace-engine/coverage-rules.ts

// C6: 实现覆盖率（对齐现有状态枚举）
export function calculateC6(rows: MatrixRow[]): number {
  const taskRows = rows.filter(r => r.id.startsWith('TASK-'));
  const implemented = taskRows.filter(r =>
    r.status === 'Implemented' || 
    r.status === 'Verified' || 
    r.status === 'Accepted'
  );
  return pct(implemented.length, taskRows.length);
}

// C7: 代码审查覆盖率（移除 PR 依赖）
export function calculateC7(rows: MatrixRow[]): number {
  const taskRows = rows.filter(r => r.id.startsWith('TASK-'));
  // 简化：有 upstream 即认为经过审查
  const reviewed = taskRows.filter(r => r.upstream.length > 0);
  return pct(reviewed.length, taskRows.length);
}
```

**关键修正**：
- ✅ C6 使用现有状态枚举（Implemented/Verified/Accepted）
- ✅ C7 移除 PR 依赖，简化为"有上游即合规"

// src/core/gate-engine/gate-evaluator.ts
export function evaluateGate(stage: string, matrix: ParsedMatrix): GateResult {
  const coverage = getCoverage(matrix);
  const integrity = checkMatrix(matrix, stage);
  
  return {
    // 1. 完整性检查
    orphans: integrity.orphans,
    brokenChains: integrity.brokenChains,
    
    // 2. 覆盖率评估（继续保留）
    'C-PRD': coverage['C-PRD'],  // PRD 覆盖率
    'C10': coverage['C10'],      // 需求规格质量
    'C2': coverage['C2'],        // FR → DS
    'C3': coverage['C3'],        // DS → TASK
    'C4': coverage['C4'],        // FR → TC
    'C6': coverage['C6'],        // 实现覆盖率
    'C9': coverage['C9'],        // 缺陷修复率
    
    // 3. 门禁判定
    passed: evaluateStageRequirements(stage, coverage, integrity),
  };
}
```
