### 1. 断链检测（✅ 已完整）

```typescript
// src/core/trace-engine/matrix.ts:58-78
export function checkMatrix(featureId: string, projectRoot: string): MatrixCheckResult {
  const brokenChains: MatrixCheckResult['brokenChains'] = [];
  for (const fr of frRows) {
    const missing: string[] = [];
    const hasPrd = (fr.upstream ?? []).some(u => u.startsWith('REQ-PRD-'));
    const hasDs = rows.some(r => r.type === 'DS' && r.upstream?.includes(fr.id));
    const hasTask = rows.some((r) => {
      if (r.type !== 'TASK') return false;
      return trace.lineage.hasAnyAncestor(r.id, new Set([fr.id]));
    });
    const hasTc = rows.some(r => r.type === 'TC' && r.upstream?.includes(fr.id));
    if (!hasPrd) missing.push('REQ-PRD-*');
    if (!hasDs) missing.push('DS');
    if (!hasTask) missing.push('TASK');
    if (!hasTc) missing.push('TC');
    if (missing.length > 0) {
      brokenChains.push({ frId: fr.id, missing });
    }
  }
  return { total: rows.length, orphans, brokenChains, vModelPairs, warnings };
}
```

**结论**：已完整检测 FR 缺少 PRD/DS/TASK/TC 的所有情况。

```typescript
// src/core/trace-engine/coverage.ts:84-91
function calcImplCoverage(taskRows: MatrixRow[]): number {
  if (taskRows.length === 0) return 1;
  const implemented = taskRows.filter(r =>
    r.status === 'Implemented' || r.status === 'Verified' || r.status === 'Accepted',
  );
  return pct(implemented.length, taskRows.length);
}

// src/core/trace-engine/coverage.ts:95-99
function calcPrCompliance(taskRows: MatrixRow[]): number {
  if (taskRows.length === 0) return 1;
  const linked = taskRows.filter(r => r.upstream && r.upstream.length > 0);
  return pct(linked.length, taskRows.length);
}
```

**结论**：
- C6: 使用 Implemented/Verified/Accepted 状态 ✅
- C7: TASK 有 upstream 的比例 ✅
