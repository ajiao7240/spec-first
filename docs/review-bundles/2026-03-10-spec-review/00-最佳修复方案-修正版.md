### Step 2: 完整的 parseMatrix

```typescript
function parseMatrixContent(content: string): MatrixRow[] {
  const rows: MatrixRow[] = [];

  for (const cells of parseMarkdownTable(content)) {
    const id = cells[0];
    const upstream = parseRefList(cells[4]);
    const downstream = parseRefList(cells[5]);

    // 1. ID 格式校验
    const validation = validateId(id);
    if (!validation.valid) {
      throw new Error(`Invalid ID: ${id} - ${validation.error}`);
    }

    // 2. 引用合法性校验（修正：我之前遗漏的）
    for (const ref of [...upstream, ...downstream]) {
      if (!validateId(ref).valid) {
        throw new Error(`Invalid reference: ${ref} in ${id}`);
      }
    }

    rows.push({
      id,
      type: validation.type!,
      title: cells[2],
      status: normalizeStatus(cells[3]),
      upstream,
      downstream,
    });
  }

  return rows;
}
```

### Step 3: 保持 checkMatrix 简单

```typescript
export function checkMatrix(
  featureId: string,
  projectRoot: string
): MatrixCheckResult {
  const rows = parseMatrix(featureId, projectRoot);
  const errors: string[] = [];

  // 只做链路完整性
  for (const row of rows) {
    const chainErrors = validateChain(row.id, row.upstream ?? []);
    errors.push(...chainErrors.map(e => `${row.id}: ${e}`));
  }

  // 孤儿检测
  const orphans = rows.filter(r =>
    r.type !== 'Feature' && r.type !== 'REQ' &&
    (!r.upstream || r.upstream.length === 0)
  );

  return {
    total: rows.length,
    errors,
    orphans,
    valid: errors.length === 0 && orphans.length === 0,
  };
}
```

### Step 4: Gate 保持完整职责

```typescript
// src/core/gate-engine/gate-evaluator.ts
export function evaluateGate(
  featureId: string,
  stage: Stage,
  projectRoot: string
): GateResult {
  const rows = parseMatrix(featureId, projectRoot);
  const coverage = getCoverage(featureId, projectRoot);
  const matrixCheck = checkMatrix(featureId, projectRoot);

  const ctx: GateContext = {
    featureId,
    stage,
    projectRoot,
    rows,
    coverage,
  };

  const conditions = GATE_CONDITIONS[stage] || [];
  const results: GateConditionResult[] = [];

  for (const cond of conditions) {
    results.push({
      id: cond.id,
      description: cond.description,
      ...cond.evaluate(ctx),
    });
  }

  // 新增：链路完整性作为前置条件
  if (!matrixCheck.valid) {
    results.unshift({
      id: 'G-MATRIX',
      description: 'Matrix integrity',
      pass: false,
      detail: matrixCheck.errors.join('; '),
    });
  }

  return {
    pass: results.every(r => r.pass),
    results,
  };
}
```

**关键修正**：
- Gate 不是"只做完整性"
- Gate = 完整性 + 覆盖率 + 质量指标 + 阶段控制
