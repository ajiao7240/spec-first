### 文件3: completeness-rules.ts（新增）

```typescript
// src/shared/completeness-rules.ts
export const COMPLETENESS_RULES = {
  FR: {
    required: {
      upstream: ['REQ'],
      downstream: ['DS', 'AC_OR_TC'],  // AC 或 TC 至少一个
    },
    optional: {
      downstream: ['TASK'],
    },
  },
  DS: {
    required: {
      upstream: ['FR'],
      downstream: ['TASK'],
    },
  },
  TASK: {
    required: {
      upstream: ['DS', 'AC'],  // DS 或 AC 至少一个
    },
  },
} as const;

export function checkCompleteness(rows: MatrixRow[]): CompletenessResult {
  const errors: string[] = [];
  const frRows = rows.filter(r => r.type === 'FR');

  for (const fr of frRows) {
    const rule = COMPLETENESS_RULES.FR;

    // 检查 upstream
    const hasReq = fr.upstream?.some(u => u.startsWith('REQ-'));
    if (!hasReq) {
      errors.push(`${fr.id}: missing REQ upstream`);
    }

    // 检查 downstream
    const hasDs = rows.some(r => r.type === 'DS' && r.upstream?.includes(fr.id));
    const hasAc = rows.some(r => r.type === 'AC' && r.upstream?.includes(fr.id));
    const hasTc = rows.some(r => r.type === 'TC' && r.upstream?.includes(fr.id));

    if (!hasDs) {
      errors.push(`${fr.id}: missing DS downstream`);
    }
    if (!hasAc && !hasTc) {
      errors.push(`${fr.id}: missing AC or TC downstream`);
    }
  }

  return {
    valid: errors.length === 0,
    errors,
  };
}
```

**修正点**：
- ✅ 新增完整性规则定义
- ✅ FR 必须映射 REQ + DS + (AC 或 TC)

```typescript
// src/core/trace-engine/id-validator.ts
export function validateChain(
  id: string,
  upstream: string[],
  downstream: string[]  // 修正：新增 downstream 参数
): string[] {
  const validation = validateId(id);
  if (!validation.valid) return ['Invalid ID format'];

  const rule = ID_RULES[validation.type!];
  const errors: string[] = [];

  // 校验 upstream
  for (const upId of upstream) {
    const upValidation = validateId(upId);
    if (!upValidation.valid) {
      errors.push(`Invalid upstream: ${upId}`);
      continue;
    }
    if (!rule.upstream.includes(upValidation.type!)) {
      errors.push(`${validation.type} cannot have ${upValidation.type} as upstream`);
    }
  }

  // 修正：校验 downstream
  for (const downId of downstream) {
    const downValidation = validateId(downId);
    if (!downValidation.valid) {
      errors.push(`Invalid downstream: ${downId}`);
      continue;
    }
    if (!rule.downstream.includes(downValidation.type!)) {
      errors.push(`${validation.type} cannot have ${downValidation.type} as downstream`);
    }
  }

  return errors;
}
```

**修正点**：
- ✅ 校验 upstream + downstream
- ✅ 双向约束完整

```typescript
// src/core/trace-engine/matrix.ts
export function checkMatrix(
  featureId: string,
  projectRoot: string
): MatrixCheckResult {
  const rows = parseMatrix(featureId, projectRoot);
  const errors: string[] = [];

  // 1. 链路合法性
  for (const row of rows) {
    const chainErrors = validateChain(
      row.id,
      row.upstream ?? [],
      row.downstream ?? []  // 修正：传入 downstream
    );
    errors.push(...chainErrors.map(e => `${row.id}: ${e}`));
  }

  // 2. 孤儿检测
  const orphans = rows.filter(r =>
    r.type !== 'Feature' && r.type !== 'REQ' && r.type !== 'RFC' &&
    (!r.upstream || r.upstream.length === 0)
  );

  // 修正：3. 完整性检查
  const completenessResult = checkCompleteness(rows);
  errors.push(...completenessResult.errors);

  return {
    total: rows.length,
    errors,
    orphans,
    valid: errors.length === 0 && orphans.length === 0,
  };
}
```

**修正点**：
- ✅ 保留完整性断链检查
- ✅ FR 缺 DS/TASK/TC 会报错

```typescript
export function evaluateGate(
  featureId: string,
  stage: Stage,
  projectRoot: string
): GateResult {
  const rows = parseMatrix(featureId, projectRoot);
  const coverage = getCoverage(featureId, projectRoot);
  const matrixCheck = checkMatrix(featureId, projectRoot);

  const ctx: GateContext = {
    featureId,
    stage,
    projectRoot,
    rows,
    coverage,
  };

  const results: GateConditionResult[] = [];

  // 前置：链路完整性
  if (!matrixCheck.valid) {
    results.push({
      id: 'G-MATRIX',
      description: 'Matrix integrity',
      pass: false,
      detail: matrixCheck.errors.join('; '),
    });
  }

  // 修正：保留质量门禁
  const conditions = GATE_CONDITIONS[stage] || [];
  for (const cond of conditions) {
    results.push({
      id: cond.id,
      description: cond.description,
      ...cond.evaluate(ctx),
    });
  }

  return {
    pass: results.every(r => r.pass),
    results,
  };
}
```
