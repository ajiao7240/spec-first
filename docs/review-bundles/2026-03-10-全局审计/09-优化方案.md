## C1: nextId 并发竞态 + 非幂等

### 问题分析

**根因**:
1. 读取 matrix → 计算序号 → 写入 matrix 三步无锁保护
2. 无幂等键，重复执行生成不同 ID
3. 无去重机制，可能写入重复 ID

**影响范围**: 所有 Artifact ID 类型 (FR/DS/TASK/TC/RFC/...)

### 最佳方案

采用 **文件锁 + 幂等键 + 去重检查** 三重保护：

#### 1. 实现文件锁机制

```typescript
// src/shared/file-lock.ts
import { openSync, closeSync, unlinkSync, existsSync } from 'fs';
import { join } from 'path';

export function withFileLock<T>(
  lockPath: string,
  action: () => T,
  options: { timeout?: number; retryInterval?: number } = {}
): T {
  const { timeout = 30000, retryInterval = 100 } = options;
  const startTime = Date.now();

  let fd: number | null = null;

  // 尝试获取锁
  while (true) {
    try {
      fd = openSync(lockPath, 'wx'); // 独占创建
      break;
    } catch (err) {
      if (Date.now() - startTime > timeout) {
        throw new Error(`获取锁超时: ${lockPath}`);
      }
      // 等待后重试
      Atomics.wait(new Int32Array(new SharedArrayBuffer(4)), 0, 0, retryInterval);
    }
  }

  try {
    return action();
  } finally {
    if (fd !== null) {
      closeSync(fd);
      if (existsSync(lockPath)) {
        unlinkSync(lockPath);
      }
    }
  }
}
```

#### 2. 重构 nextId 函数

```typescript
// src/core/trace-engine/id-generator.ts

export interface NextIdOptions {
  type: NextIdType;
  abbr: string;
  featureId: string;
  projectRoot: string;
  tcLevel?: string;
  requestKey?: string; // 新增：幂等键
}

export function nextId(opts: NextIdOptions): NextIdResult {
  const lockPath = join(
    opts.projectRoot,
    'specs',
    opts.featureId,
    '.matrix.lock'
  );

  return withFileLock(lockPath, () => {
    // 1. 校验缩写格式
    validateAbbr(opts.abbr);

    const matrixPath = getMatrixPath(opts.projectRoot, opts.featureId);

    // 2. 检查幂等键
    if (opts.requestKey) {
      const existing = findByRequestKey(matrixPath, opts.requestKey);
      if (existing) {
        return { id: existing.id, seq: existing.seq, reused: true };
      }
    }

    // 3. 读取现有 ID
    const rows = parseMatrixIds(matrixPath);

    // 4. 计算下一个序号
    const seq = findNextSeq(rows, opts.type, opts.abbr, opts.tcLevel);

    // 5. 组装 ID
    const id = assembleId(opts.type, opts.abbr, seq, opts.tcLevel);

    // 6. 校验 ID 格式
    const validation = validateId(id);
    if (!validation.valid) {
      throw new Error(`生成了无效 ID: ${id} - ${validation.error}`);
    }

    // 7. 去重检查
    if (rows.some(row => row.id === id)) {
      throw new Error(`ID 已存在: ${id}`);
    }

    // 8. 写入矩阵
    appendToMatrix(matrixPath, {
      id,
      type: opts.type,
      title: '',
      status: 'Planned',
      requestKey: opts.requestKey
    });

    return { id, seq, reused: false };
  });
}
```

```typescript
// src/core/ai-orchestrator/todo-runner.ts

export function loadTodoState(
  projectRoot: string,
  featureId: string
): TodoRunnerState | null {
  const path = getTodoStatePath(projectRoot, featureId);

  if (!existsSync(path)) {
    return null;

  try {
    const content = readFileSync(path, 'utf-8');
    const obj = JSON.parse(content);

    if (!isTodoRunnerState(obj)) {

    // 版本兼容性检查
    if (obj.version !== '1.0') {
      throw new Error(`不支持的 TodoRunnerState 版本: ${obj.version}`);
    }

    return obj;
  } catch (err) {
    throw new Error(`加载 todo-state 失败: ${err.message}`);
  }
}
```

## H1: TASK 解析规则碎片化

### 问题分析

**根因**:
- 7 处不同的 TASK 解析逻辑

```typescript
// src/core/ai-orchestrator/task-plan-parser.ts

export function parseTaskPlan(content: string): TaskPlan {
  const items: TaskPlanItem[] = [];
  let currentTask: string | undefined;

  // 支持两种格式：表格和 checkbox
  if (content.includes('|')) {
    // 表格格式
    items.push(...parseTableFormat(content));
  } else {
    // Checkbox 格式
    items.push(...parseCheckboxFormat(content));
  }

  // 查找当前任务
  currentTask = items.find(item => item.status === 'in_progress')?.taskId;

  return { items, currentTask };
}

function parseTableFormat(content: string): TaskPlanItem[] {
  const items: TaskPlanItem[] = [];
  const lines = content.split('\n');

  for (const line of lines) {
    if (!line.includes('TASK-')) continue;

    const match = line.match(/\|\s*(TASK-[\w-]+)\s*\|\s*([^|]+)\s*\|\s*([^|]+)\s*\|/);
    if (!match) continue;

    const [, taskId, title, statusRaw] = match;
    const status = normalizeStatus(statusRaw.trim());

    items.push({ taskId, title: title.trim(), status });
  }

  return items;
}

function parseCheckboxFormat(content: string): TaskPlanItem[] {
  const items: TaskPlanItem[] = [];
  const lines = content.split('\n');

  for (const line of lines) {
    const match = line.match(/^-\s*\[([ x])\]\s*(TASK-[\w-]+):\s*(.+)$/);
    if (!match) continue;

    const [, checkbox, taskId, title] = match;
    const status = checkbox === 'x' ? 'completed' : 'pending';

    items.push({ taskId, title, status });
  }

  return items;
}

function normalizeStatus(raw: string): TaskPlanItem['status'] {
  const normalized = raw.toLowerCase().replace(/[_\s-]/g, '');

  if (['inprogress', '进行中', 'doing'].includes(normalized)) {
    return 'in_progress';
  }
  if (['completed', '已完成', 'done'].includes(normalized)) {
    return 'completed';
  }
  if (['blocked', '阻塞', '受阻'].includes(normalized)) {
    return 'blocked';
  }
  return 'pending';
}
```

## H2: Feature ID 解析不统一

### 问题分析

**根因**:
- `listFeatures()` 取自 `stage-state.json.featureId`
- `getFeatureState()` 按目录路径读取
- 未校验"目录名 == stage-state.featureId == current"

**影响**: 目录名与状态文件不一致时出现错乱

### 最佳方案

采用 **三元组一致性校验 + 统一解析入口** 策略：

#### 1. 实现三元组校验

```typescript
// src/core/process-engine/feature.ts

export interface FeatureTriple {
  dirName: string;
  stateFeatureId: string;
  currentPointer: string | null;
}

export function validateFeatureTriple(
  projectRoot: string,
  featureId: string
): FeatureTriple {
  const dirName = featureId;
  const dirPath = join(projectRoot, 'specs', dirName);

  // 1. 检查目录存在
  if (!existsSync(dirPath)) {
    throw new Error(`Feature 目录不存在: ${dirName}`);
  }

  // 2. 读取 stage-state.json
  const statePath = join(dirPath, 'stage-state.json');
  if (!existsSync(statePath)) {
    throw new Error(`stage-state.json 不存在: ${dirName}`);
  }

  const state = JSON.parse(readFileSync(statePath, 'utf-8'));
  const stateFeatureId = state.featureId;
```

```typescript
export function resolveFeatureId(
  projectRoot: string,
  input: string
): string {
  const features = listFeatures(projectRoot);

  // Exact match
  if (features.includes(input)) {
    validateFeatureTriple(projectRoot, input);
    return input;
  }

  // Prefix match
  const matches = features.filter(f => f.startsWith(input));
  if (matches.length === 1) {

    return matches[0];
  }
  if (matches.length > 1) {
    throw new Error(`多个 Feature 匹配前缀 "${input}": ${matches.join(', ')}`);
  }

  throw new Error(`未找到 Feature: ${input}`);
}
```

## H3: retry-controller 未接入

### 问题分析

**根因**:
- `makeRetryDecision()` 已定义但未调用
- auto-loop 失败分支未接入 retry 逻辑

**影响**: retry budget、backoff 保护未生效

### 最佳方案

```typescript
// src/core/ai-orchestrator/auto-loop.ts

export async function runAutoLoop(
  featureId: string,
  projectRoot: string,
  options: AutoLoopOptions
): Promise<AutoLoopResult> {
  let state = loadTodoState(projectRoot, featureId) || initTodoState(options);

  while (state.iteration < state.maxIterations && !state.halted) {
    updateHeartbeat(state);

    const readyTodos = pickReadyTodos(state);
    if (readyTodos.length === 0) {
      state.halted = true;
      break;
    }

    const todo = readyTodos[0];

    try {
      const result = await executeTodo(todo, featureId, projectRoot);

      if (result.success) {
        markTodoCompleted(state, todo.id);
      } else {
        // 接入 retry-controller
        const retryDecision = makeRetryDecision(
          state,
          todo.id,
          result.error,
          classifyFailure(result.error)
        );

        if (retryDecision.shouldRetry) {
          // 应用 retry 状态
          applyRetryToState(state, todo.id, retryDecision);
          
          // Backoff 等待
          if (retryDecision.backoffMs > 0) {
            await sleep(retryDecision.backoffMs);
          }
        } else {
          // 预算用尽或不可重试
          markTodoBlocked(state, todo.id, result.error);
          
          if (options.stopOnBlocked) {
            state.halted = true;
            break;
          }
        }
      }
    } catch (err) {
      // 未预期的异常
      markTodoBlocked(state, todo.id, err.message);
      
      if (options.stopOnBlocked) {
        state.halted = true;
        break;
      }
    }

    // Watchdog 检查
    const watchdogResult = runWatchdogCheck(state);
    if (watchdogResult.shouldHalt) {
      state.halted = true;
      break;
    }

    state.iteration++;
    saveTodoState(projectRoot, featureId, state);
  }

  return {
    completed: state.items.filter(t => t.status === 'completed').length,
    blocked: state.items.filter(t => t.status === 'blocked').length,
    halted: state.halted
  };
}

function classifyFailure(error: string): FailureType {
  if (error.includes('timeout')) return 'timeout';
  if (error.includes('rate limit')) return 'rate_limit';
  if (error.includes('network')) return 'transient';
  return 'permanent';
}
```

```typescript
// src/core/process-engine/advance.ts

export function advance(
  featureId: string,
  projectRoot: string,
  options: AdvanceOptions = {}
): AdvanceResult {
  const lockPath = join(projectRoot, 'specs', featureId, '.state.lock');

  return withFileLock(lockPath, () => {
    // 1. 加载当前状态
    const state = loadState(featureId, projectRoot);
    const from = state.currentStage;

    // 2. 计算下一阶段
    const to = nextStageInChain(from);

    // 3. 执行 Gate 校验
    const gateResult = evaluateGate(featureId, projectRoot, to);
    if (!gateResult.passed && !options.force) {
      throw new Error(`Gate 校验失败: ${gateResult.reason}`);
    }

    // 4. 更新状态
    state.currentStage = to;
    state.history.push({
      from,
      to,
      timestamp: new Date().toISOString(),
      gateResult
    });

    // 5. 保存状态
    saveState(featureId, projectRoot, state);

    // 6. 同步矩阵状态
    syncMatrixStatus(featureId, projectRoot, to);

    return { from, to, gateResult };
  });
}

function syncMatrixStatus(
  featureId: string,
  projectRoot: string,
  stage: Stage
): void {
  const matrixPath = getMatrixPath(projectRoot, featureId);
  const rows = parseMatrix(matrixPath);

    const expectedStatus = getStatusForStage(stage, row.type);
    if (row.status !== expectedStatus) {
      updateMatrixRow(matrixPath, row.id, { status: expectedStatus });
    }
  }
}
```

```typescript
// src/core/ai-orchestrator/audit-log.ts

export function writeAuditLog(
  projectRoot: string,
  featureId: string,
  entry: AuditEntry
): void {
  const auditPath = getAuditLogPath(projectRoot, featureId);
  const lockPath = `${auditPath}.lock`;

  withFileLock(lockPath, () => {
    const prevHash = getLastHash(auditPath);
    const record: AuditRecord = {
      ...entry,
      timestamp: new Date().toISOString(),
      prevHash,
      hash: computeHash(entry, prevHash)
    };

  });
}

export function readAuditLog(
  projectRoot: string,
  featureId: string
): AuditRecord[] {
  const auditPath = getAuditLogPath(projectRoot, featureId);
  if (!existsSync(auditPath)) return [];

  const records: AuditRecord[] = [];
  const lines = readFileSync(auditPath, 'utf-8').split('\n');
  const corruptedLines: number[] = [];

  for (let i = 0; i < lines.length; i++) {
    const line = lines[i].trim();
    if (!line) continue;

    try {
      const record = JSON.parse(line);
      records.push(record);
    } catch (err) {
      corruptedLines.push(i + 1);
    }
  }

  // 显式报告损坏行
  if (corruptedLines.length > 0) {
    throw new Error(
      `审计日志包含 ${corruptedLines.length} 行损坏数据: 行号 ${corruptedLines.join(', ')}`
    );
  }

  return records;
}

export function verifyAuditChain(
  projectRoot: string,
  featureId: string
): { valid: boolean; errors: string[] } {
  const records = readAuditLog(projectRoot, featureId);
  const errors: string[] = [];

  for (let i = 0; i < records.length; i++) {
    const record = records[i];
    const expectedPrevHash = i === 0 ? 'genesis' : records[i - 1].hash;

    if (record.prevHash !== expectedPrevHash) {
      errors.push(
        `记录 ${i}: prevHash 不匹配 (期望 ${expectedPrevHash}, 实际 ${record.prevHash})`
      );
    }

    const computedHash = computeHash(record, record.prevHash);
    if (record.hash !== computedHash) {
      errors.push(`记录 ${i}: hash 校验失败`);
    }
  }

  return { valid: errors.length === 0, errors };
}
```

```typescript
export function appendToMatrix(
  matrixPath: string,
  row: MatrixRow
): void {
  const tmpPath = `${matrixPath}.tmp-${process.pid}`;

  try {
    // 1. 读取现有内容
    const content = existsSync(matrixPath)
      ? readFileSync(matrixPath, 'utf-8')
      : '| ID | Type | Title | Status |\n|-----|------|-------|--------|\n';

    const newContent = content + newRow;

    // 3. 写入临时文件
    writeFileSync(tmpPath, newContent, 'utf-8');

    // 4. 原子重命名
    renameSync(tmpPath, matrixPath);
  } catch (err) {
    // 清理临时文件
    if (existsSync(tmpPath)) {
      unlinkSync(tmpPath);
    }
    throw err;
  }
}
```

## M3: 循环依赖静默处理

### 问题分析

**根因**: 检测到循环返回空集合，不报错

**影响**: 用户无法发现循环依赖

### 最佳方案

```typescript
// src/core/trace-engine/upstream-lineage.ts

export function createUpstreamLineage(rows: MatrixRow[]): UpstreamLineage {
  const upstreamMap = new Map<string, string[]>();
  const cycles: Array<{ cycle: string[] }> = [];

  // 构建上游映射
  for (const row of rows) {
    upstreamMap.set(row.id, row.upstreamIds || []);
  }

  const getAncestors = (
    id: string,
    stack = new Set<string>()
  ): ReadonlySet<string> => {
    // 检测循环
    if (stack.has(id)) {
      const cycleArray = Array.from(stack);
      const startIdx = cycleArray.indexOf(id);
      const cycle = [...cycleArray.slice(startIdx), id];
      cycles.push({ cycle });
      
      // 抛出异常而非返回空集合
      throw new Error(
        `检测到循环依赖: ${cycle.join(' → ')}`
      );
    }

    const ancestors = new Set<string>();
    const upstreams = upstreamMap.get(id) || [];

    for (const upstreamId of upstreams) {
      ancestors.add(upstreamId);
      
      const upstreamAncestors = getAncestors(
        upstreamId,
        new Set([...stack, id])
      );
      
      for (const ancestor of upstreamAncestors) {
        ancestors.add(ancestor);
      }
    }

    return ancestors;
  };

  return {
    getAncestors,
    detectCycles: () => cycles
  };
}
```

```typescript
// src/shared/file-lock.ts

export interface FileLockOptions {
  timeout?: number;        // 获取锁超时时间（ms）
  retryInterval?: number;  // 重试间隔（ms）
}

export function withFileLock<T>(
  lockPath: string,
  action: () => T,
  options: FileLockOptions = {}
): T {
  const { timeout = 30000, retryInterval = 100 } = options;
  const startTime = Date.now();
  let fd: number | null = null;

  // 尝试获取锁
  while (true) {
    try {
      fd = openSync(lockPath, 'wx');
      break;
    } catch (err) {
      if (Date.now() - startTime > timeout) {
        throw new Error(`获取锁超时: ${lockPath}`);
      }
      Atomics.wait(new Int32Array(new SharedArrayBuffer(4)), 0, 0, retryInterval);
    }
  }

  try {
    return action();
  } finally {
    if (fd !== null) {
      closeSync(fd);
      if (existsSync(lockPath)) {
        unlinkSync(lockPath);
      }
    }
  }
}
```

```typescript
  version: string;
  validate: (obj: unknown) => obj is T;
}

export function createSchemaGuard<T>(
  kind: string,
  version: string,
  validate: (obj: unknown) => obj is T
): SchemaGuard<T> {
  return { kind, version, validate };
}

export function loadWithSchemaGuard<T>(
  path: string,
  guard: SchemaGuard<T>
): T | null {
  if (!existsSync(path)) return null;

  const content = readFileSync(path, 'utf-8');
  const obj = JSON.parse(content);

  if (!guard.validate(obj)) {
    throw new Error(
      `Schema 校验失败: 期望 ${guard.kind}@${guard.version}, 实际 ${obj.kind}@${obj.version}`
    );
  }

  return obj;
}
```
