## 一、ID 生成最佳实践

### 1.1 使用锁保护并发生成

**原则**: 所有 ID 生成操作必须加锁

```typescript
// ✅ 正确做法
export function nextId(opts: NextIdOptions): NextIdResult {
  return withMatrixLock(opts.projectRoot, opts.featureId, () => {
    // 读取 → 计算 → 写入 在锁内完成
    const rows = parseMatrixIds(matrixPath);
    const seq = findNextSeq(rows, opts.type, opts.abbr);
    const id = assembleId(opts.type, opts.abbr, seq);
    appendToMatrix(matrixPath, { id, ... });
    return { id, seq };
  });

// ❌ 错误做法
export function nextId(opts: NextIdOptions): NextIdResult {

  const seq = findNextSeq(rows, opts.type, opts.abbr);

}
```

### 2.2 事务性更新

**原则**: 多个文件更新要么全成功，要么全失败

```typescript
// ✅ 正确做法
export function advance(...): AdvanceResult {
  const backup = {
    state: loadState(featureId, projectRoot),
    matrix: readMarkdown(matrixPath)
  };

  try {
    state.currentStage = to;
    saveState(featureId, projectRoot, state);
    syncMatrixStatus(featureId, projectRoot, to);
    return { from, to };
  } catch (error) {
    // 回滚
    saveState(featureId, projectRoot, backup.state);
    writeMarkdown(matrixPath, backup.matrix);
    throw error;
  }
}
```

function parseMatrixContent(content: string): MatrixRow[] {
  const rows: MatrixRow[] = [];
  for (const cells of parseMarkdownTable(content)) {
    const id = cells[0];
    const validation = validateId(id);
    if (!validation.valid) {
      throw new Error(`矩阵中存在无效 ID：${id} - ${validation.error}`);

## 五、监控最佳实践

### 5.1 ID 重复检测

**原则**: 定期扫描矩阵，检测重复 ID

```typescript
export function detectDuplicateIds(featureId: string, projectRoot: string): string[] {
  const rows = parseMatrix(featureId, projectRoot);
  const idCounts = new Map<string, number>();

  for (const row of rows) {
    idCounts.set(row.id, (idCounts.get(row.id) || 0) + 1);
  }

  return Array.from(idCounts.entries())
    .filter(([_, count]) => count > 1)
    .map(([id]) => id);
}
```

export function checkStateConsistency(featureId: string, projectRoot: string): string[] {
  const state = loadState(featureId, projectRoot);
  const rows = parseMatrix(featureId, projectRoot);
  const warnings: string[] = [];

  for (const row of rows) {
    const expectedStatus = getStatusForStage(state.currentStage, row.type);
    if (row.status !== expectedStatus) {
      warnings.push(`${row.id}: 期望 ${expectedStatus}，实际 ${row.status}`);
    }
  }

  return warnings;
}
