> **版本**: 5.0
> **日期**: 2026-03-27
> **状态**: Phase-aware + 规则引擎 + LLM 增强
> **核心理念**: 规则引擎保证基础质量和性能，LLM 提供智能增强

---

## 概述

本方案采用**混合架构**，结合规则引擎的可靠性和 LLM 的灵活性：

**核心架构**:

```text
规则引擎（基础层）
  - 快速、可靠、一致
  - 处理标准场景
  ↓
Phase-aware 选择
  - 每个 phase 预计算并存储
  - 响应执行过程变化
  ↓
双模式注入
  - 模式A: 动态注入（Claude/iFlow）
  - 模式B: 静态生成（Codex/Kiro）
```

**总工作量**:

- P1 核心功能: 25-32 小时
- P2 平台适配: 6-8 小时
- P3 未来扩展: 待定

---

## 核心原则

1. **Phase-aware** - 每个 phase 预计算并存储 skills（可追溯）
2. **规则优先** - 标准场景用规则引擎（快速、可靠、确定性）
3. **双模式架构** - 动态注入（Claude/iFlow）vs 静态生成（Codex/Kiro）
4. **单向数据流** - Skills资产从源到平台的清晰流向
5. **可追溯** - 记录每个 phase 的 selector_result 和 selector_trace

```python
def update_phase(task_dir: str, new_phase: int, new_mode: str = None):
    """更新任务的 phase"""
    task_json = Path(task_dir) / "task.json"
    task_data = json.loads(task_json.read_text())

    # 记录当前 phase 到历史
    if task_data.get('current_phase'):
        history_entry = {
            "phase": task_data['current_phase'],
            "started_at": task_data.get('phase_started_at', datetime.now().isoformat()),
            "completed_at": datetime.now().isoformat(),
            "task_mode": task_data['execution_profile'].get('task_mode'),
            "selector_result": task_data.get('selector_result', {})
        }
        task_data.setdefault('phase_history', []).append(history_entry)

        # Phase 历史上限保护（防止数据膨胀）
        MAX_PHASE_HISTORY = 100
        if len(task_data['phase_history']) > MAX_PHASE_HISTORY:
            task_data['phase_history'] = task_data['phase_history'][-MAX_PHASE_HISTORY:]

    # 更新到新 phase
    task_data['current_phase'] = new_phase
    task_data['phase_started_at'] = datetime.now().isoformat()

    # 可选：切换 task_mode
    if new_mode:
        task_data['execution_profile']['task_mode'] = new_mode

    # 重新计算 selector_result
    from .skill_selector import select_skills
    selector_result = select_skills(
        execution_profile=task_data['execution_profile'],
        phase=new_phase,
        registry_path='.spec-first/config/skills-registry.json'
    )
    task_data['selector_result'] = selector_result

    task_json.write_text(json.dumps(task_data, indent=2))
```

```python
def select_skills(
    execution_profile: dict,
    phase: int,
    registry_path: str
) -> dict:
    """
    Phase-aware skill selection (规则引擎)

    Args:
        execution_profile: 任务执行环境
        phase: 当前 phase 编号（整数，与现有代码一致）
        registry_path: registry 文件路径

    Returns:
        {
            "selected_skills": [...],
            "selector_trace": {
                "matched": [...],
                "filtered": [...],
                "conflicts_resolved": [...],
                "final": [...]
            }
        }
    """
    registry = _load_registry(registry_path)
    rules = _load_rules(registry_path.replace('skills-registry', 'selector-rules'))

def _load_registry(registry_path: str):
    """加载 registry，支持用户覆盖"""
    # 1. 尝试用户配置
    user_config = Path('.spec-first/config/skills-registry.json')
    if user_config.exists():
        return json.loads(user_config.read_text())

    # 2. Fallback 到默认配置
    default_config = Path('.spec-first/defaults/skills-registry.json')
    return json.loads(default_config.read_text())

def _load_rules(rules_path: str):
    """加载 rules，支持用户覆盖"""
    user_config = Path('.spec-first/config/selector-rules.json')
    if user_config.exists():
        return json.loads(user_config.read_text())

    default_config = Path('.spec-first/defaults/selector-rules.json')
    return json.loads(default_config.read_text())

    task_mode = execution_profile.get('task_mode', 'default')
    surface = execution_profile.get('surface')
    language = execution_profile.get('language')

    # Step 1: 匹配 phase bundle
    bundle_skills = rules.get('phase_bundles', {}).get(str(phase), [])

    # Step 2: 过滤适用的 skills
    matched = []
    for skill in registry['skills']:
        if _is_applicable(skill, phase, task_mode, surface, language):
            matched.append(skill)

    # Step 3: 按 priority 排序
    matched.sort(key=lambda s: s.get('priority', 0), reverse=True)

    # Step 4: 解决冲突
    selected = _resolve_conflicts(matched, rules)

    # Step 5: 添加 requires
    selected = _add_required_skills(selected, registry)

    # Step 6: 应用配额
    selected = _apply_quotas(selected, rules)

    return {
        "selected_skills": [s['id'] for s in selected],
        "selector_trace": {
            "bundle": bundle_skills,
            "matched": [s['id'] for s in matched],
            "final": [s['id'] for s in selected]
        }
    }

def _is_applicable(skill, phase, task_mode, surface, language):
    """检查 skill 是否适用（规则匹配）"""
    applicable = skill.get('applicable_to', {})

    # 1. 检查 phase（最重要）
    phases = applicable.get('phases', [])
    if phases and phase not in phases:
        return False

    # 2. 检查 task_mode
    modes = applicable.get('task_modes', [])
    if modes and task_mode not in modes:
        return False

    # 3. 检查 surface
    surfaces = applicable.get('surfaces', [])
    if surfaces and surface not in surfaces:
        return False

    # 4. 检查 language
    languages = applicable.get('languages', [])
    if languages and language not in languages:
        return False

    return True

def _resolve_conflicts(skills, rules):
    """解决 conflicts_with"""
    selected = []
    excluded_ids = set()

    for skill in skills:
        if skill['id'] in excluded_ids:
            continue
        selected.append(skill)
        # 排除冲突的 skills
        for conflict_id in skill.get('conflicts_with', []):
            excluded_ids.add(conflict_id)

    return selected

def _add_required_skills(skills, registry):
    """添加 requires 依赖"""
    skill_map = {s['id']: s for s in registry['skills']}
    result = list(skills)

    for skill in skills:
        for req_id in skill.get('requires', []):
            if req_id in skill_map and skill_map[req_id] not in result:
                result.append(skill_map[req_id])

    return result

def _apply_quotas(skills, rules):
    """应用 category_quotas"""
    quotas = rules.get('category_quotas', {})
    # 简化实现：按 priority 截断
    return skills
```

```python
# .spec-first/scripts/validate_registry.py
def validate_skills_registry(registry_path: str) -> dict:
    """验证 skills-registry.json 的一致性

    Returns:
        {
            "valid": bool,
            "errors": list[str],
            "warnings": list[str]
        }
    """
    registry = json.loads(Path(registry_path).read_text())
    errors = []
    warnings = []

    skill_ids = {s['id'] for s in registry['skills']}

    for skill in registry['skills']:
        skill_id = skill['id']

        # 1. 检查 requires 引用
        for req_id in skill.get('requires', []):
            if req_id not in skill_ids:
                errors.append(f"{skill_id}: requires 引用不存在的 skill '{req_id}'")

        # 2. 检查 conflicts_with 引用
        for conflict_id in skill.get('conflicts_with', []):
            if conflict_id not in skill_ids:
                errors.append(f"{skill_id}: conflicts_with 引用不存在的 skill '{conflict_id}'")

        # 3. 检查 priority 范围
        priority = skill.get('priority', 50)
        if not (0 <= priority <= 100):
            warnings.append(f"{skill_id}: priority {priority} 超出推荐范围 [0-100]")

        # 4. 检查循环依赖
        if _has_circular_dependency(skill_id, skill, registry['skills']):
            errors.append(f"{skill_id}: 存在循环依赖")

    return {
        "valid": len(errors) == 0,
        "errors": errors,
        "warnings": warnings
    }

def _has_circular_dependency(skill_id: str, skill: dict, all_skills: list, visited=None) -> bool:
    """检查是否存在循环依赖"""
    if visited is None:
        visited = set()

    if skill_id in visited:
        return True

    visited.add(skill_id)
    skill_map = {s['id']: s for s in all_skills}

    for req_id in skill.get('requires', []):
        if req_id in skill_map:
            if _has_circular_dependency(req_id, skill_map[req_id], all_skills, visited.copy()):
                return True

    return False
```

```python
def select_skills_hybrid(
    execution_profile: dict,
    phase: int,
    user_context: str = None,
    timeout: int = 5
) -> dict:
    """混合选择策略（带缓存和 fallback）"""

    # Layer 1: 规则引擎基础选择（快速、可靠）
    base_result = select_skills(execution_profile, phase, registry_path)

    # Layer 2: LLM 调整（仅在有特殊上下文时）
    if not user_context:
        return base_result

    # 检查缓存
    cache_key = _hash_context(user_context, base_result)
    cached = _get_cache(cache_key)
    if cached:
        return cached

    try:
        # LLM 调整（带超时）
        adjusted = _adjust_by_llm(
            base_result['selected_skills'],
            user_context,
            execution_profile,
            timeout=timeout
        )

        # 验证 LLM 结果
        if _validate_llm_result(adjusted, base_result):
            result = {
                "selected_skills": adjusted['skills'],
                "selector_trace": {
                    **base_result['selector_trace'],
                    "llm_adjusted": True,
                    "llm_action": adjusted['action']
                }
            }
            _set_cache(cache_key, result)
            return result
        else:
            # LLM 结果无效，fallback 到规则引擎
            return base_result

    except TimeoutError:
        # 超时 fallback
        return base_result
    except Exception as e:
        # 异常 fallback
        logging.warning(f"LLM adjustment failed: {e}")
        return base_result
```
